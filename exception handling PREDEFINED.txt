CREATE OR REPLACE PROCEDURE EXPDEMO(P_EID NUMBER)

IS

V_ENAME EMPLOYEE.ENAME%TYPE;

	BEGIN 	
		
		SELECT ENAME INTO V_ENAME FROM EMPLOYEE WHERE EID = P_EID;
		
		DBMS_OUTPUT.PUT_LINE ('UR NAME IS ' || V_ENAME);
	EXCEPTION 
		
		WHEN NO_DATA_FOUND THEN
		
		DBMS_OUTPUT.PUT_LINE ('SORRY INVALID EMPLOYEE ID');
		
	END ;
