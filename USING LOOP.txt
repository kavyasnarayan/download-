CREATE OR REPLACE PROCEDURE C_DEMO 

IS

V_EID NUMBER;
 
V_ENAME EMPLOYEE.ENAME%TYPE;

V_SAL NUMBER;

CURSOR C1 IS SELECT EID ,ENAME,SALARY FROM EMPLOYEE;

BEGIN

OPEN C1;

LOOP

FETCH C1 INTO V_EID,V_ENAME,V_SAL; 

DBMS_OUTPUT.PUT_LINE(V_EID ||' ' || V_ENAME|| ' '|| V_SAL);

EXIT WHEN C1%NOTFOUND;

END LOOP;

CLOSE C1;

END;
